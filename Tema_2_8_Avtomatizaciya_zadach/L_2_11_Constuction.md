1. Переменные и типы данных
1.1. Что такое переменная
Переменная — это именованный контейнер для хранения данных. Можно представить как коробку с наклейкой, где лежит какое-то значение.

python
# Простейший пример
name = "Анна"      # в переменную name помещаем строку "Анна"
age = 25           # в переменную age помещаем число 25
height = 1.68      # в переменную height помещаем число с плавающей точкой
is_student = True  # в переменную is_student помещаем логическое значение

# Использование переменных
print(name)        # Анна
print(age + 5)     # 30

1.5. Преобразование типов (Type Casting)
python
# Преобразование в целое число (int)

1.6. Динамическая типизация
Python не требует явного указания типа переменной. Тип определяется автоматически и может меняться.

python
# Динамическая типизация
value = 10          # int
print(type(value))  # <class 'int'>

value = "Hello"     # теперь это str
print(type(value))  # <class 'str'>

value = 3.14        # теперь это float
print(type(value))  # <class 'float'>

# Это удобно, но нужно быть внимательным
# Лучше избегать неявной смены типов в одной переменной

2.3. Ввод нескольких значений
python
# Способ 1: через split()
data = input("Введите два числа через пробел: ")
a, b = data.split()            # разделяем строку на части
a = int(a)
b = int(b)
print(f"Сумма: {a + b}")

# Способ 2: более компактно
a, b = map(int, input("Введите два числа: ").split())
print(f"Сумма: {a + b}")

# Способ 3: ввод списка
numbers = list(map(int, input("Введите числа через пробел: ").split()))
print(f"Сумма всех чисел: {sum(numbers)}")

# lower() / upper() — преобразование регистра
answer = input("Вы согласны? (да/нет): ").lower().strip()

3. Вывод данных (print)
3.1. Базовое использование
python
# Вывод строки
print("Hello, World!")

# Вывод нескольких значений
name = "Анна"
age = 25
print("Имя:", name, "Возраст:", age)   # print автоматически добавляет пробелы

# Вывод с разделителем (sep)
print("a", "b", "c", sep="-")          # a-b-c
print(1, 2, 3, sep=", ")               # 1, 2, 3

# Вывод без перевода строки (end)
print("Загрузка", end=" ")
print("...", end=" ")
print("Готово!")                       # Загрузка ... Готово!

# Пустая строка
print()                                 # просто перевод строки
3.2. Параметры sep и end
python
# Параметр sep (separator) — разделитель между значениями
print("apple", "banana", "cherry", sep=", ")   # apple, banana, cherry
print(1, 2, 3, sep="")                         # 123
print("a", "b", "c", sep="\n")                 # a\nb\nc (каждый на новой строке)

# Параметр end (концовка строки)
print("Hello", end="")
print("World")                                 # HelloWorld
print("First", end=" -> ")
print("Second", end=" -> ")
print("Third")                                 # First -> Second -> Third

# Пример: индикатор загрузки
import time
print("Загрузка", end="")
for i in range(5):
    print(".", end="", flush=True)
    time.sleep(0.5)
print(" Готово!")
4. Форматирование вывода
4.1. Конкатенация строк (старый способ)
python
name = "Анна"
age = 25

# Конкатенация (склеивание) строк
message = "Меня зовут " + name + ", мне " + str(age) + " лет."
print(message)
# Недостатки: нужно явно преобразовывать числа в строки
4.2. Форматирование через % (как в C)
python
name = "Анна"
age = 25
height = 1.68

print("Меня зовут %s, мне %d лет, мой рост %.2f м" % (name, age, height))
# %s — строка
# %d — целое число
# %f — вещественное число (.2f — 2 знака после запятой)
4.3. Метод format() (современный способ)
python
name = "Анна"
age = 25
height = 1.68

# По порядку
print("Меня зовут {}, мне {} лет".format(name, age))

# По индексу
print("{0} учится в {1} классе. {0} любит математику.".format("Анна", 10))

# По имени
print("Меня зовут {name}, мне {age} лет".format(name="Анна", age=25))

# С форматированием чисел
print("Число π = {:.3f}".format(3.14159265))        # 3.142
print("В процентах: {:.1%}".format(0.25))           # 25.0%
print("С разделением тысяч: {:,}".format(1000000))  # 1,000,000
4.4. f-строки (лучший способ, Python 3.6+)
python
name = "Анна"
age = 25
height = 1.68

# f-строки — самый удобный и читаемый способ
print(f"Меня зовут {name}, мне {age} лет")

# Выражения внутри фигурных скобок
print(f"Через 5 лет мне будет {age + 5} лет")
print(f"Квадрат числа {age} равен {age ** 2}")

# Форматирование чисел
print(f"Число π = {3.14159265:.3f}")      # 3.142
print(f"Процент: {0.25:.1%}")             # 25.0%
print(f"Большое число: {1000000:,}")      # 1,000,000
print(f"С выравниванием: {'текст':^20}")  #         текст

# Многострочные f-строки
message = f"""
Пользователь: {name}
Возраст: {age}
Рост: {height} м
Статус: {'Студент' if age < 25 else 'Работает'}
"""
print(message)
4.5. Сравнение способов форматирования
python
name = "Анна"
age = 25

# Способ 1: конкатенация (устарел, неудобен)
print("Меня зовут " + name + ", мне " + str(age) + " лет.")

# Способ 2: % (работает, но громоздкий)
print("Меня зовут %s, мне %d лет." % (name, age))

# Способ 3: format() (хороший, универсальный)
print("Меня зовут {}, мне {} лет.".format(name, age))

# Способ 4: f-строки (лучший, Python 3.6+)
print(f"Меня зовут {name}, мне {age} лет.")
Рекомендация: Используйте f-строки — они самые читаемые и производительные.
