Задание: Напишите Python-модуль date_calculator.py, который содержит:

Функцию days_between(date1, date2) — возвращает количество дней между двумя датами

Функцию add_workdays(start_date, days) — добавляет к дате указанное количество рабочих дней (пн-пт)

Функцию get_week_number(date) — возвращает номер недели в году

Используйте модуль datetime. Добавьте docstring и аннотации типов.

Эталонное решение:

python
"""
date_calculator.py — Модуль для расчёта дат и временных интервалов.
"""

from datetime import date, timedelta
from typing import Tuple


def days_between(date1: date, date2: date) -> int:
    """
    Возвращает количество дней между двумя датами.

    Args:
        date1: Первая дата
        date2: Вторая дата

    Returns:
        Абсолютное количество дней между датами
    """
    return abs((date2 - date1).days)


def add_workdays(start_date: date, days: int) -> date:
    """
    Добавляет к дате указанное количество рабочих дней.

    Args:
        start_date: Начальная дата
        days: Количество рабочих дней для добавления

    Returns:
        Новая дата
    """
    current = start_date
    added = 0
    
    while added < days:
        current += timedelta(days=1)
        # weekday(): 0=пн, 4=пт, 5=сб, 6=вс
        if current.weekday() < 5:
            added += 1
    
    return current


def get_week_number(date_obj: date) -> int:
    """
    Возвращает номер недели в году.

    Args:
        date_obj: Дата

    Returns:
        Номер недели (1-53)
    """
    return date_obj.isocalendar()[1]


# Пример использования
if __name__ == "__main__":
    d1 = date(2024, 1, 1)
    d2 = date(2024, 12, 31)
    
    print(f"Дней между {d1} и {d2}: {days_between(d1, d2)}")
    print(f"Через 5 рабочих дней от {d1}: {add_workdays(d1, 5)}")
    print(f"Номер недели для {d1}: {get_week_number(d1)}")

Нарушения SOLID:

Принцип	Нарушение
SRP	Класс отвечает за: хранение данных, валидацию, сохранение в файл, отправку email
OCP	Чтобы добавить новое хранилище (БД), нужно менять код класса
DIP	Зависит от конкретных реализаций (файл, email), а не от абстракций
Рефакторинг:

python
from abc import ABC, abstractmethod

# Абстракции
class UserRepository(ABC):
    @abstractmethod
    def save(self, user): pass

class EmailService(ABC):
    @abstractmethod
    def send_welcome(self, email): pass

class UserValidator(ABC):
    @abstractmethod
    def validate_email(self, email): pass

# Реализации
class FileUserRepository(UserRepository):
    def save(self, user):
        with open("users.txt", "a") as f:
            f.write(f"{user.name},{user.email}\n")

class ConsoleEmailService(EmailService):
    def send_welcome(self, email):
        print(f"Welcome email sent to {email}")

class EmailValidator(UserValidator):
    def validate_email(self, email):
        if '@' not in email:
            raise ValueError("Invalid email")
        return True

# Сервис
class UserService:
    def __init__(self, repo: UserRepository, email_service: EmailService, validator: UserValidator):
        self.repo = repo
        self.email_service = email_service
        self.validator = validator
