2.2. Относительные импорты (только внутри пакета)
Относительный импорт использует точки для указания пути относительно текущего файла.

python
# Внутри файла my_package/math_utils.py

# Импорт из соседнего модуля (в той же папке)
from .string_utils import capitalize

# Импорт из родительского пакета
from ..network.http_client import send_request

# Импорт из пакета на уровень выше
from ...config import settings
Правила использования точек:

Синтаксис	Значение
.	текущий пакет
..	родительский пакет
...	на два уровня выше
Важное правило: относительные импорты работают только внутри пакета и только в модулях, которые импортированы как часть пакета (не в скриптах, запущенных напрямую).

2.3. Сравнение: что когда использовать
Ситуация	Использовать
Главные файлы проекта (main.py, app.py)	Абсолютные импорты
Модули внутри пакета, которые не будут перемещаться	Абсолютные импорты
Модули внутри пакета, которые могут быть переименованы	Относительные (легче рефакторить)
Тесты (папка tests/)	Абсолютные (проще и надёжнее)

Старый способ (os.path) работает, но громоздкий:

Современный способ (pathlib) — объектно-ориентированный и кроссплатформенный:

from pathlib import Path

absolute = path.absolute()
dirname = path.parent

3.3. Практический пример: обход проекта
python
from pathlib import Path

def count_py_lines(project_root):
    """Считает количество строк кода во всех .py файлах проекта."""
    root = Path(project_root)
    total_lines = 0
    py_files = list(root.glob('**/*.py'))  # ** — рекурсивно
    
    for file in py_files:
        lines = len(file.read_text(encoding='utf-8').splitlines())
        total_lines += lines
        print(f"{file.relative_to(root)}: {lines} строк")
    
    print(f"\nВсего: {total_lines} строк в {len(py_files)} файлах")
    return total_lines

# Использование

python
import json
from pathlib import Path
from models.user import User

class UserRepository:
    """Абстракция над хранилищем пользователей."""
    
    def __init__(self, data_path: Path):
        self.data_path = data_path
    
    def save(self, user: User) -> bool:
        """Сохраняет пользователя в JSON-файл."""
        data = {'name': user.name, 'email': user.email}
        with open(self.data_path, 'a', encoding='utf-8') as f:
            f.write(json.dumps(data) + '\n')
        return True
    
    def find_by_email(self, email: str) -> User | None:
        """Ищет пользователя по email."""
        if not self.data_path.exists():
            return None
        
        for line in self.data_path.read_text().splitlines():
            data = json.loads(line)
            if data['email'] == email:
                return User(data['name'], data['email'])
        return None

from datetime import datetime
from dataclasses import dataclass, asdict

@dataclass
class Note:
    """Модель заметки (только данные)."""
    title: str
    content: str
    created_at: datetime = None
    
    def __post_init__(self):
        if self.created_at is None:
            self.created_at = datetime.now()
    
    def to_dict(self):
        return asdict(self)
    
    @staticmethod
    def from_dict(data: dict):
        return Note(
            title=data['title'],
            content=data['content'],
            created_at=datetime.fromisoformat(data['created_at'])
        )

import json
from pathlib import Path
from typing import List, Optional
from src.models.note import Note

class NoteRepository:
    """Репозиторий для работы с заметками в JSON-файле."""
    
    def __init__(self, file_path: Path):
        self.file_path = file_path
        self._ensure_file_exists()
    
    def _ensure_file_exists(self):
        if not self.file_path.exists():
            self.file_path.parent.mkdir(parents=True, exist_ok=True)
            self.file_path.write_text('[]', encoding='utf-8')
    
    def _read_all(self) -> List[dict]:
        content = self.file_path.read_text(encoding='utf-8')
        return json.loads(content) if content else []
    
    def _write_all(self, data: List[dict]):
        self.file_path.write_text(json.dumps(data, indent=2, ensure_ascii=False), encoding='utf-8')
    
    def save(self, note: Note) -> Note:
        notes = self._read_all()
        notes.append(note.to_dict())
        self._write_all(notes)
        return note
    
    def get_all(self) -> List[Note]:
        notes_dict = self._read_all()
        return [Note.from_dict(n) for n in notes_dict]
    
    def find_by_title(self, title: str) -> Optional[Note]:
        for note in self.get_all():
            if note.title.lower() == title.lower():
                return note
        return None

from typing import List, Optional
from src.models.note import Note
from src.repositories.note_repo import NoteRepository

class NoteService:
    """Сервис с бизнес-логикой для работы с заметками."""
    
    def __init__(self, repository: NoteRepository):
        self.repository = repository
    
    def create_note(self, title: str, content: str) -> Note:
        if not title or not content:
            raise ValueError("Заголовок и содержание не могут быть пустыми")
        
        existing = self.repository.find_by_title(title)
        if existing:
            raise ValueError(f"Заметка с заголовком '{title}' уже существует")
        
        note = Note(title=title, content=content)
        return self.repository.save(note)
    
    def list_notes(self) -> List[Note]:
        return self.repository.get_all()
    
    def find_note(self, title: str) -> Optional[Note]:
        return self.repository.find_by_title(title)
